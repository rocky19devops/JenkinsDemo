pipeline {
  agent any
  stages {
    stage('Checkout') {
      steps {
        sh 'echo passed'
        //git branch: 'main', url: 'git@github.com:rocky19devops/JenkinsDemo.git'
      }
    }
    stage('Test contents') {
      steps {
        sh 'ls -ltrh'
      }
    }
    stage('build') {
      steps {
        sh 'cd newapp/newapp && docker build -t rockynewapps:latest .'
        sh 'docker images'
      }
    }
    stage('Deploy') {
      steps {
        sh 'docker run -it --name rockynewapps rockynewapps .'
      }
    }
   stage('Testing the application') {
      steps {
        sh 'curl -v http://$HOSTNAME:9090'
      }
    }
  }
}
